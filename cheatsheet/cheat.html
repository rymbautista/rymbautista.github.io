HTML

 
      <div class="imageBox">
      <a href="./bio/bio.html">
      <div class="imageInn">
      <img class="picture"
        src="/images/IMG_7888.png" alt="Default Image">
      </div>
      <div class="hoverImg">
        <img class="picture"
        src="/images/Untitled_Artwork (3) copy.png" alt="Profile Image">

        <span class="caption">About Me!

      </div>

CSS

 
    .fade-in-text {
      font-family: Lora;
      font-size: 50px;
      padding-top: 15px;
      text-align: center;
      }
      
      .fade-in-text:hover {
        animation: fadeIn 5s;
        -webkit-animation: fadeIn 5s;
        -moz-animation: fadeIn 5s;
        -o-animation: fadeIn 5s;
        -ms-animation: fadeIn 5s;
        }
        
        @keyframes fadeIn {
        0% { opacity: 0; }
        100% { opacity: 1; }
        }
        
        @-moz-keyframes fadeIn {
        0% { opacity: 0; }
        100% { opacity: 1; }
        }
        
        @-webkit-keyframes fadeIn {
        0% { opacity: 0; }
        100% { opacity: 1; }
        }
        
        @-o-keyframes fadeIn {
        0% { opacity: 0; }
        100% { opacity: 1; }
        }
        
        @-ms-keyframes fadeIn {
        0% { opacity: 0; }
        100% { opacity: 1; }
        }
